	cin >> s;
	for (int i = 0; i < n; i++) {
		if (s[i]=='V'&& s[i + 1]=='K') {
			cnt += 1;
			s[i] = 0;
			s[i + 1] = 1;
		}
	}
	for (int i = 0; i < n; i ++ ){
		if (s[i]!=0&&s[i] == s[i + 1]) {
			cnt += 1;
			break;
		}
	}
	cout << cnt;
