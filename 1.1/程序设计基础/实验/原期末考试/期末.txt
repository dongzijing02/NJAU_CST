void paixu() {
	int i = 0, j, k, t;
	int s = 19222126; int a[10] = { 0 };
	while (s) {
		a[i++] = s % 10; s /= 10;
	}
	for (j = 0; j < i - 1; j++) {
		for (k = 0; k < i - 1 - j; k++) {
			if (a[k] > a[k + 1]) {
				t = a[k + 1]; a[k + 1] = a[k]; a[k] = t;
			}
		}
	}
	for (j = i - 1; j >= 0; j--) {
		printf_s("%d", a[j]);
	}
	printf_s("\n");
}
void wanmei() {
	int i, j, t, k, a[5000];
	for (i = 1; i <= 10000; i++) {
		k = 0; t = 0;
		for (j = 1; j < i; j++) {
			if (i % j == 0) {
				t += j; a[k++] = j;
			}
		}
		if (t == i) {
			printf("%d=%d", i, a[0]);
			for (j = 1; j < k; j++) {
				printf("+%d", a[j]);
			}
			printf("\n");
		}
	}
}
void find_b(char s[]) {
	int i, t = 0;
	for (i = 0; s[i] != '\0'; i++) {
		if (s[i] == ' ' && s[i + 1] == 'b')t++;
	}
	if (t != 0)printf("%d", t);
	else printf("不存在");
	printf("\n");
}
